// Function to render songs
function renderSongs(songs) {
    const songBody = document.getElementById('song-body');
    songBody.innerHTML = '';
    songs.forEach(song => {
        const row = document.createElement('tr');
        row.innerHTML = `
            <td>${song.name}</td>
            <td>${song.artist}</td>
            <td>${song.year}</td>
        `;
        songBody.appendChild(row);
    });
}

// Render songs initially
renderSongs(songs);

// Add event listeners for sorting buttons
document.getElementById('name-asc').addEventListener('click', () => {
    songs.sort((a, b) => a.name.localeCompare(b.name));
